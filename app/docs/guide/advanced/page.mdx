end
subgraph BB [Services]

end

  const rawJs = await compileMdx(`${mermaidCodeblock}
## Usage
~~~md filename="Markdown"

~~~
`)
  return <MDXRemote compiledSource={rawJs} components={{ Mermaid }} />
}
